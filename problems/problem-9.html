Project Euler with Javascript

Problem 9 description

A Pythagorean triplet is a set of three natural numbers, a < b < c, for which,

a2 + b2 = c2

For example, 32 + 42 = 9 + 16 = 25 = 52.

There exists exactly one Pythagorean triplet for which a + b + c = 1000.

Find the product abc.

My solution

			
	function problem9(number) {
		var	answer = 0;
		var a = 0;
		var b = 0;
		var c = 0;

		// a < b < c, so it stands to reason that a only goes up to a third of the sum of a, b and c
		for ( a = 1; a <= number / 3; a++ ) { 

			// Similarly, b could potentially be close to half to still be smaller than c and bigger than a. 
			// Odds are we'll find the answer before then.
			for ( b = a + 1; b < number / 2; b++) { 

				// c will be the remainder to make it easier and use less variables
				c = number - a - b;	

				// If c is larger than 0 and the three make a Pythagorean triplet, we can get the product
				if ( c > 0 && (( a * a ) + ( b * b ) == ( c * c )) ) {	
					answer = a * b * c;
					return answer;
				}
			}	
		}
	}

	problem9(1000);
